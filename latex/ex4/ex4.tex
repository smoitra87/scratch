\paragraph{Combining with the algorithm package to get a nice box}
\begin{algorithm}
\caption{Euclid's Algorithm} \label{algo:euclid}
\begin{algorithmic}[1]
\Procedure{Euclid}{$a,b$}\Comment{The g.c.d of a and b}
	\State $r \gets a \mod b$
	\While{$r\neq 0$} \Comment{Keep iterating till $r=0$}
		\State $a \gets b$
		\State $b \gets r$
		\State $r \gets a \mod b$
	\EndWhile
	\State \textbf{return} $b$ \Comment{The solution is $b$}
\EndProcedure
\Statex \Comment{Statex used to insert gap}
\Procedure{Fac}{$n$}\Comment{Calculates factorial of n}
	\If{$n==1$}
		\State {\bf return} $1$
	\Else
		\State {\bf return} $n*$\Call{Fac}{$n-1$}\Comment{Call to call functions}
	\EndIf
\EndProcedure
\end{algorithmic}
\end{algorithm}

\section{Algorithm Continuation}
In this section we will see how wecan break and then continue an algorithm

\begin{algorithm}
\caption{Euclid's Algorithm} \label{algo:euclid}
\begin{algorithmic}[1]
\Procedure{Euclid}{$a,b$}\Comment{The g.c.d of a and b}
	\State $r \gets a \mod b$
	\While{$r\neq 0$} \Comment{Keep iterating till $r=0$}
		\State $a \gets b$
\algstore{eucbreak}
\end{algorithmic}
\end{algorithm}


\begin{algorithm}
\caption{Euclid's algorithm continued} \label{algo:euclid2}
\begin{algorithmic}[1]
\algrestore{eucbreak}
		\State $b \gets r$
		\State $r \gets a \mod b$
	\EndWhile
	\State \textbf{return} $b$ \Comment{The solution is $b$}
\EndProcedure

\end{algorithmic}
\end{algorithm}
